            /**
             * Checks whether a passed time string is in format HH:MM:SS
             * @param {string} timeString
             * @return {bool}
             */
            function isValidTimeFormat(timeString) {
                timeParts = timeString.split(':');
                hours = timeParts[0];
                minutes = timeParts[1];
                seconds = timeParts[2];

                return (hours < 24) && (minutes < 60) && (seconds < 60);
            }
